## 赋值转化
## 切片
``` python
array_2d = np.array([[18, 26, 17], 
                     [25, 15.5, 12], 
                     [24, 27, 20],
                     [10, 5.5, 17],
                     [27, 26, 15],
                     [22, 21, 21]])
print(array_2d[0, 1])          # 某一个值 先行后列
--- 26     
print(array_2d[3: , ])         # 连续的几列
---
[10, 5.5, 17],
[27, 26, 15],
[22, 21, 21]])
print(array_2d[[0, 2, 1]])     # 分开的几列 要用list
---
[[18.  26.  17. ]
 [24.  27.  20. ]
 [25.  15.5 12. ]]
print(array_2d[:, [0,2]])      # 分开的几行 要用list 且前面要加,
---
[[18. 17.]
 [25. 12.]
 [24. 20.]
 [10. 17.]
 [27. 15.]
 [22. 21.]]
```
