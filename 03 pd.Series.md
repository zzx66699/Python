``` python
import pandas as pd
import numpy as np
nums = pd.Series([1, 8, 7, 5, 6, 5, 3, 4, 7, 1])

import pandas as pd
from datetime import datetime 
year = 2019
start = datetime(year,1,1)  # 从datetime模块调用的
end = datetime(year,12,31)
result = pd.Series(pd.date_range(start=start, end=end, freq='W-SUN'))
print(result)
```
